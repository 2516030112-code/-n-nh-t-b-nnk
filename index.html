<script>

    function datMon(tenMon, giaTien) {

        // Hiện khu vực thanh toán

        document.getElementById('checkout').style.display = 'block';

       

        // Điền dữ liệu vào Form gửi Mail

        document.getElementById('formMonAn').value = tenMon;

        document.getElementById('formGia').value = giaTien.toLocaleString('vi-VN') + ' đ';

       

        // CẤU HÌNH NGÂN HÀNG CỦA BẠN TẠI ĐÂY

        var bankId = "MB"; // Thay bằng tên viết tắt ngân hàng của bạn (VCB, TCB, ACB, BIDV, ICB...)

        var stk = "123456789"; // Thay bằng Số tài khoản ngân hàng chính xác của bạn

        var noidung = "SAKURA " + tenMon.replace(/\s/g, ''); // Tạo nội dung viết liền không dấu

       

        // Tạo link API tự động sinh mã VietQR

        var qrUrl = "https://img.vietqr.io/image/" + bankId + "-" + stk + "-compact2.jpg?amount=" + giaTien + "&addInfo=" + encodeURIComponent(noidung);

       

        // Thay thế hiển thị trên web

        document.getElementById('vietqr').src = qrUrl;

        document.getElementById('ckNoidung').innerText = noidung;

       

        // Cuộn màn hình mượt mà xuống phần điền thông tin

        document.getElementById('checkout').scrollIntoView({ behavior: 'smooth' });

    }

</script>
